webpack初步的搭建与配置

环境搭建

1.全局安装

cnpm i -g webpack

2.本地安装

cnpm i webpack --save-dev

3.开始打包

webpack ./main.js(主文件路径) bundle.js(出口文件路径)

配置

配置文件 webpack.config.js

1.引入webpack模块
    var webpack = require('webpack');
2.导出配置对象
    //导出配置对象
    module.exports={
        //声明入口主文件
        entry:'./js/main.js',
        //声明出口文件(最终生成的文件名)
        output:{
            //声明出口文件的路径
            path: './output/',
            //声明出口文件名
            filename:'bundle.js'
        },
        //声明模块加载器
        module:{
            //声明具体的模块加载器
            loaders:[
                //css加载器:让所有的以.css结尾的文件都使用css加载器进行解析
                {test:/\.css$/,loader:'style-loader!css-loader'},
                {test:/\.less$/,loader:'style!css!less'}
            ]
        }
    };
