ES5的继承写法

//父类
function Father(){
    //相应的属性代码
}
//父类的原型方法
Father.prototype.show=function(){
    //方法代码
}
//子类
function Son(){
     //继承第一句:让子类拥有父类的属性
     Father.apply(this,arguments);
    //子类相应的属性代码
}
//继承第二句:子类继承父类的原型方法
Son.prototype=Object.create(Father.prototype);
//继承第三句:找回丢失的构造函数
Son.prototype.constructor=Son;
//子类的原型方法
Son.prototype.run=function(){
    //方法代码
}

ES6的继承写法

class Son extends Father{
    constructor(){
        super();
    }
}
